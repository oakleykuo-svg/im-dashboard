// 渲染核心 function render(data){ const members={}, stages={}; let total=0, exCount=0, activeDataCount=0; data.forEach(item => { // 1. 取得歸檔標籤 (來自 V5.0 GAS 腳本) const isArchive = item["isArchive"] === true; const n = item[KEYS.NAME] || 'Unknown'; const s = parseFloat(item[KEYS.SCORE]) || 0; const st = item[KEYS.STATUS] || 'Other'; // 2. 只有「非歸檔」的案子，才計入當前 Loading 指數與執行中件數 if (!isArchive) { total += s; if(st === 'Executing') exCount++; if(!members[n]) members[n] = {score:0, count:0}; members[n].score += s; members[n].count++; activeDataCount++; } // 3. 但無論是否歸檔，都計入階段統計 (為了月/季度的報表完整性) // 只有在全部(All)或符合時間過濾的情況下才會執行 stages[st] = (stages[st] || 0) + 1; }); const mCount = Object.keys(members).length; const avg = mCount ? Math.round(total/mCount) : 0; // 更新上方 KPI 數字 (只顯示當前負載) countUp(document.getElementById('kpiTotal'), Math.round(total)); countUp(document.getElementById('kpiExecuting'), exCount); countUp(document.getElementById('kpiAvg'), avg); countUp(document.getElementById('kpiCount'), activeDataCount); // 顯示當前活躍專案數 document.getElementById('memberBadge').textContent = mCount + ' members'; document.getElementById('stageBadge').textContent = data.length + ' projects'; // 總量含 Archive // --- 以下圖表與排行榜繪製邏輯保持不變 --- // 個人產值 Bar Chart const sorted = Object.entries(members).sort((a,b) => b[1].score - a[1].score); const labels = sorted.map(([n]) => n); const vals = sorted.map(([,v]) => Math.round(v.score)); const bCols = labels.map((_,i) => PALETTE[i % PALETTE.length]); if(mChart) mChart.destroy(); mChart = new Chart(document.getElementById('memberChart').getContext('2d'), { type: 'bar', data: { labels, datasets:[{ label:'產值指數', data:vals, backgroundColor: bCols.map(c=>c+'28'), borderColor: bCols, borderWidth:2, borderRadius:9 }]}, options: { responsive: true, plugins: { legend: { display: false } }, scales: { x: { grid: { display: false } }, y: { beginAtZero: true, grid: { color: 'rgba(0,0,0,0.05)' } } } } }); // 專案階段 Doughnut (這裡會顯示含 Archive 的總分佈) const sL = STAGE_ORDER.filter(s => stages[s]); const sV = sL.map(s => stages[s]); const sC = sL.map(s => STAGE_COLORS[s]); if(sChart) sChart.destroy(); sChart = new Chart(document.getElementById('stageChart').getContext('2d'), { type: 'doughnut', data: { labels: sL, datasets: [{ data: sV, backgroundColor: sC.map(c=>c+'33'), borderColor: sC, borderWidth: 2.5 }]}, options: { cutout: '72%', plugins: { legend: { display: false } } } }); // 階段 Legend 渲染 const totalS = sV.reduce((a,b) => a+b, 0); document.getElementById('stageLegend').innerHTML = sL.map((s,i) => `
${s}
${sV[i]}
`).join(''); // 排行榜細節 (只顯示當前活躍成員) const listEl = document.getElementById('memberList'); listEl.innerHTML = ''; sorted.forEach(([name, val], i) => { const pct = Math.min((val.score/THRESHOLD)*100, 100); let cls = pct > 85 ? 'danger' : (pct > 55 ? 'warning' : 'ok'); const row = document.createElement('div'); row.innerHTML = `
${name[0]}
${name}
${val.count} 個活躍專案
${Math.round(val.score)}PTS
${cls === 'danger' ? 'OVERLOAD' : (cls === 'warning' ? 'HIGH' : 'HEALTHY')}
`; listEl.appendChild(row); }); }
